                messages.success(request, u_message)
                user.activate = True
                user.phone_verified = user.phone
                user.save()

                from sales.send_messages import send_welcome_email
                send_welcome_email(user, request, u_message)

                return redirect('homepage-home')
            else:
                u_message = _('Incorrect PassCode!')
                messages.success(request, u_message)
        form = PassCodeForm()
        context = {"form": form}
        return render(request, self.template_name, context)

[docs]    def get(self, request):
        try:
            received_list = CMessage.objects.filter(receiver=self.request.user.profile)
            sent_list = CMessage.objects.filter(sender=self.request.user.profile)

            unread_received_count = received_list.filter(read=False, init=True).count()
            unread_sent_count = sent_list.filter(read=False, init=False).count()
            total_unread = unread_received_count + unread_sent_count

            unread_received_senders = received_list.filter(read=False, init=True).distinct('sender')
            unread_sent_senders = sent_list.filter(read=False, init=False).distinct('receiver')
            all_unread_senders = unread_received_senders.union(unread_sent_senders)

            receiver_list = received_list.distinct('sender')
            sender_list = sent_list.distinct('receiver')
            all_messages = receiver_list.union(sender_list)
        except Exception as e:
            print(e)
            messages.warning(request, _('Setup your Profile to use messages'))
            return redirect('profile-create')
        context = {'all_messages': all_messages.order_by('date_posted'), 'total_unread': total_unread, 'all_unread_senders': all_unread_senders}
        return render(request, self.template_name, context)
