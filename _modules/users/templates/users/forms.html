class UserRegisterForm(UserCreationForm):
    first_name = forms.CharField
    last_name = forms.CharField
    email = forms.EmailField
    user_type = _("User")
    company_type = _("Company")
    c = [("1", user_type), ("2", company_type)]
    regtype = forms.ChoiceField(choices=c, label=_("Register as"))

    class Meta:
        model = User
        fields = ['username', 'first_name', 'last_name', 'email', 'password1', 'password2', 'regtype']

    def clean_national_id(self, *args, **kwargs):
        cleaned_national_id = self.cleaned_data.get('national_id')
        if self.instance.country == 'mozambique':
            result = re.findall('[A]\d{8}$', cleaned_national_id)
            if not result or not len(cleaned_national_id) == 9:
                raise forms.ValidationError(_('Invalid National ID - Mozambique - A12345678'))
        elif self.instance.country == 'kenya':
            result = re.findall('\d{8}', cleaned_national_id)
            if not result or not 8 <= len(cleaned_national_id) <= 9:
                raise forms.ValidationError(_('Invalid National ID - Kenya - 12345678/9'))
        elif self.instance.country == 'angola':
            result = re.findall('\d{9}$', cleaned_national_id)
            if not result or not len(cleaned_national_id) == 9:
                raise forms.ValidationError(_('Invalid National ID - Angola - 123456789'))
        else:
            result = re.findall('\d{8}$', cleaned_national_id)
            if not result:
                raise forms.ValidationError(_('Invalid National ID - Other'))
        return cleaned_national_id

    def clean_tax_id(self, *args, **kwargs):
        cleaned_tax_id = self.cleaned_data.get('tax_id')
        if self.instance.country == 'mozambique':
            result = re.findall('[A-Z]\d{9}[A-Z]', cleaned_tax_id)
            if not result or not len(cleaned_tax_id) == 11:
                raise forms.ValidationError(_('Invalid Tax ID - Mozambique - A123456789A'))
        elif self.instance.country == 'kenya':
            result = re.findall('[A-Z]\d{9}[A-Z]', cleaned_tax_id)
            if not result or not len(cleaned_tax_id) == 11:
                raise forms.ValidationError(_('Invalid Tax ID - Kenya - A123456789A'))
        elif self.instance.country == 'angola':
            result = re.findall('[A-Z]\d{9}[A-Z]', cleaned_tax_id)
            if not result or not len(cleaned_tax_id) == 11:
                raise forms.ValidationError(_('Invalid Tax ID - Angola - A123456789A'))
        else:
            result = re.findall('[A-Z]\d{9}[A-Z]', cleaned_tax_id)
            if not result:
                raise forms.ValidationError(_('Invalid Tax ID - Other'))
        return cleaned_tax_id

    def clean_image(self):
        cleaned_image = self.cleaned_data.get('image')

                return cleaned_image

                raise forms.ValidationError(_('Only .jpg or .jpeg allowed'))
