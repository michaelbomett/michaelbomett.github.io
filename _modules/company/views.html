    return render(request, 'company/home.html', context)

        obj = get_object_or_404(CompProfile, pk=self.kwargs['pk'])
        try:
            if self.request.user.compprofile == obj:
                if EmailAddress.objects.filter(user=self.request.user, email=obj.email, verified=True).exists():
                    context['verified'] = True
                else:
                    context['verified'] = False

                my_supply = get_object_or_404(Profile, user=self.request.user)
                context['my_approved'] = my_supply.approved.all()
                if not obj.geometry:
                    if self.request.user.profile.country == 'kenya':
                        location = ['36.821967885986396', '-1.2901525291272793']
                    elif self.request.user.profile.country == 'mozambique':
                        location = ['32.5732', '-25.9692']
                    elif self.request.user.profile.country == 'angola':
                        location = ['32.70664714999998', '-25.922400671873504']
                    else:
                        location = ['32.5732', '-25.9692']

                    pnt = Point(float(location[0]), float(location[1]), srid=4326)
                    obj.geometry = pnt
                    obj.save()
                context['waypoints'] = obj
            else:
                context['waypoints'] = obj

        except Exception as e:
            print(e)
        return context
