class CompProfileCreateForm(forms.ModelForm):
    company_name = forms.CharField(max_length=50, label=_('Company Name'))
    phone = forms.CharField(max_length=13, widget=forms.TextInput(attrs={'placeholder': '+ international number'}), label=_('Phone'))
    email = forms.EmailField(help_text=_('A valid email address, please.'))
    pin = forms.CharField(max_length=20, label=_('PIN'))
    address = forms.CharField(max_length=200, label=_('Address'))
    website = forms.CharField(max_length=200, required=False, widget=forms.TextInput(attrs={'placeholder': 'http://'}), label=_('Website'))
    image = forms.ImageField(required=False, label=_('image (.jpg) 4:3 aspect ratio'))

    class Meta:
        model = CompProfile
        fields = ['company_name', 'address', 'pin', 'phone', 'email', 'website', 'image', ]

    def clean_pin(self, *args, **kwargs):
        cleaned_pin = self.cleaned_data.get('pin')
        if self.instance.country == 'mozambique':
            result = re.findall('[A]\d{8}$', cleaned_pin)
            if not result or not len(cleaned_pin) == 9:
                raise forms.ValidationError(_('Invalid PIN - Mozambique - A12345678'))
        elif self.instance.country == 'kenya':
            result = re.findall('\d{8}', cleaned_pin)
            if not result or not 8 <= len(cleaned_pin) <= 9:
                raise forms.ValidationError(_('Invalid PIN - Kenya - 12345678/9'))
        elif self.instance.country == 'angola':
            result = re.findall('\d{9}$', cleaned_pin)
            if not result or not len(cleaned_pin) == 9:
                raise forms.ValidationError(_('Invalid PIN - Angola - 123456789'))
        else:
            result = re.findall('\d{8}$', cleaned_pin)
            print(result)
            if not result:
                raise forms.ValidationError(_('Invalid PIN - Other'))
        return cleaned_pin

    def clean_image(self):
        cleaned_image = self.cleaned_data.get('image')
        if cleaned_image:
            if cleaned_image.name.endswith('.jpg') or cleaned_image.name.endswith('.jpeg'):
                return cleaned_image
            else:
                raise forms.ValidationError(_('Only .jpg or .jpeg allowed'))
        else:
            return cleaned_image

class ProductForm(forms.ModelForm):

    product_name = forms.CharField(max_length=100, label=_('Product Name'))
    product_units = forms.ChoiceField(choices=UNITS, label=_('Units'))
    product_price = forms.DecimalField(max_digits=100, decimal_places=2, label=_('Price'))
    product_description = forms.CharField(max_length=500, required=False, label=_('Description'))
    product_category = forms.ModelMultipleChoiceField(queryset=SubCategory.objects.all(), required=False, label=_('Category'))
    product_code = forms.CharField(max_length=30, required=False, label=_('Code'))
    activate = forms.BooleanField(required=False, label=_('Active'))
    tax_exemption = forms.BooleanField(required=False, label=_('Tax Exempt'))
    audience = forms.ChoiceField(choices=AUDIENCE)
    image = forms.ImageField(required=False, label=_('image (.jpg) 4:3 aspect ratio'))

    class Meta:
        model = Pricing
        fields = ('product_name', 'product_units', 'product_price', 'product_code', 'product_category', 'product_description', 'tax_exemption',
                  'exemption', 'audience', 'activate', 'image'
                  )

    def __init__(self, country, *args, **kwargs):
        super(ProductForm, self).__init__(*args, **kwargs)
        self.fields['exemption'] = forms.ModelChoiceField(Exemptions.objects.filter(country=country), label=_('Exemption'), required=False)
