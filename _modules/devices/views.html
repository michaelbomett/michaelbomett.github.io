[docs]def send_invoice_fcm(invoice):

    invoice_no = str(invoice.id)
    if invoice.created_by.profile.my_company_name:
        supplier = str(invoice.created_by.profile.my_company_name)
    else:
        supplier = str(invoice.created_by.profile)
    amount = str(invoice.total)
    company_to = invoice.company.user
    inv_url = 'https://techfactura.com/sales/payments/invoices/' + invoice_no + '/'
    image_url = str(invoice.created_by.profile.image.url)
    inv_date = str(date.today())
    devices = FCMDevice.objects.filter(user=company_to)
    try:
        if company_to.settings.push_notifications:
            for device in devices:
                device.send_message(title=_('New Invoice') + ' No: ' + invoice_no,
                                    click_action=inv_url,
                                    icon=image_url,
                                    body=_('From: ') + supplier + '\n' + _('Amount: ') + amount,
                                    data={
                                        'Image': image_url,
                                        'Invoice': invoice_no,
                                        'From': supplier,
                                        'Amount': amount,
                                        'Date': inv_date
                                    })
    except Exception as e:
        print(e)




[docs]def send_payment_fcm(payment):

    payment_no = str(payment.id)
    supplier = payment.seller.user
    company_from = str(payment.company)
    amount = str(payment.payment_value)

    if payment.payment_type == 'mpesa':
        payment_type = _('MPesa')
    elif payment.payment_type == 'cash':
        payment_type = _('Cash')
    elif payment.payment_type == 'cheque':
        payment_type = _('Cheque')
    elif payment.payment_type == 'bank_deposit':
        payment_type = _('Bank Deposit')
    else:
        payment_type = _('Other')

    pay_url = 'https://techfactura.com/sales/payments/receipts/' + payment_no + '/'
    image_url = str(payment.company.image.url)
    pay_date = str(date.today())
    devices = FCMDevice.objects.filter(user=supplier)

    try:
        if supplier.settings.push_notifications:
            for device in devices:
                device.send_message(title=_('New') + ' ' + _('Payment No: ') + payment_no,
                                    icon=image_url,
                                    click_action=pay_url,
                                    body=_('From: ') + company_from + '\n' + _('Amount: ') + amount + '\n' + _('Paid via: ') + payment_type,
                                    data={
                                        'Image': image_url,
                                        'Payment': payment_no,
                                        'From': company_from,
                                        'Amount': amount,
                                        'Paid_via': payment_type,
                                        'Date': pay_date
                                    })
    except Exception as e:
        print(e)
