class ConfirmPasswordForm(forms.ModelForm):
    confirm_password = forms.CharField(widget=forms.PasswordInput())

    class Meta:
        model = User
        fields = ('confirm_password', )

    def clean(self):
        cleaned_data = super(ConfirmPasswordForm, self).clean()
        confirm_password = cleaned_data.get('confirm_password')
        if not check_password(confirm_password, self.instance.password):
            self.add_error('confirm_password', _('Password does not match.'))

    def save(self, commit=True):
        user = super(ConfirmPasswordForm, self).save(commit)
        user.last_login = timezone.now()
        if commit:
            user.save()
        return user

class UserProductForm(forms.ModelForm):

    product_name = forms.CharField(max_length=100, label=_('Advert Name'))
    product_units = forms.ChoiceField(choices=UNITS, label=_('Units'))
    product_price = forms.DecimalField(max_digits=100, decimal_places=2, label=_('Price'))
    product_description = forms.CharField(max_length=500, required=False, widget=forms.Textarea, label=_('Description'))
    product_category = forms.ModelMultipleChoiceField(queryset=SubCategory.objects.all(), required=False, label=_('Category'))
    product_code = forms.CharField(max_length=30, required=False, label=_('Code'))
    tax_exemption = forms.BooleanField(required=False, label=_('Tax Inclusive'))
    delivery = forms.CharField(max_length=500, required=False, label=_('Delivery'))
    delivery_fee = forms.DecimalField(max_digits=100, decimal_places=2, required=False, label=_('Delivery Fee'))
    product_status = forms.ChoiceField(choices=PRODUCT_STATUS, label=_('Condition'))
    image = forms.ImageField(required=False, label=_('image (.jpg) 4:3 aspect ratio'))

    class Meta:
