account_sid = keys.ACCOUNT_SID

[docs]def send_payment_messages(payinvoice, paytype, amount, reference, request):
    """ send sms and email payment notifications"""

    company = str(payinvoice.company)
    invoice_id = str(payinvoice.id)
    invoice_total = str(payinvoice.total)
    supplier_email = payinvoice.created_by.email
    supplier = payinvoice.created_by
    payment_pk = payinvoice.id
    image_url = str(payinvoice.company.image.url)

    if supplier.settings.sms_notifications:
        if payinvoice.created_by.profile.phone_verified:
            phone = str(payinvoice.created_by.profile.phone_verified)

            try:
                sms = client.messages.create(
                    from_=keys.FROM_PHONE,
                    body=_('Company') + ': ' + company + ' \n' + _('Invoice') + ' No: ' + invoice_id + '\n' +
                          _('Payment Type') + ': ' + paytype + '\n' + _('Amount') + ': ' +
                          amount + ' \n' + _('Reference') + ': ' + reference,
                    to=phone
                )
                messages.success(request, _('SMS sent: ') + sms.status)
            except Exception as e:
                print(e)
                messages.warning(request, _('Error: SMS was not sent'))
        else:
            messages.warning(request, _('Supplier has no verified Phone number'))

    if supplier.settings.email_notifications:
        if EmailAddress.objects.filter(user=supplier, verified=True).exists():
            html_message = loader.render_to_string(
                'sales/payment_email.html',
                {
                    'user_name': supplier.profile,
                    'company': company,
                    'invoice_no': invoice_id,
                    'payment_type': paytype,
                    'payment_value': amount,
                    'invoice_total': invoice_total,
                    'reference': reference,
                    'payment_pk': payment_pk,
                    'image_url': image_url
                }
            )

            try:
                send_mail(_('Payment from ') + company,
                          _('Invoice Number: ') + invoice_id + '\n' + _('Invoice Amount: ') + invoice_total + '\n' + _('Payment type: ') +
                          paytype + '\n' + _('Payment value: ') + amount +
                          '\n' + _('Reference: ') + reference + '\n' + _('This is an automatic email'),
                          from_email,
                          [supplier_email],
                          fail_silently=True,
                          html_message=html_message
                          )
                messages.success(request, _('Email sent successfully'))
            except BadHeaderError:
                messages.warning(request, _('Error: Email was not sent'))
        else:
            messages.warning(request, _('Supplier has no verified Email Address'))

                      html_message=html_message
